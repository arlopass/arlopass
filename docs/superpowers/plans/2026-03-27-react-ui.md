# @byom-ai/react-ui Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build `@byom-ai/react-ui` — headless React component primitives for AI chat interfaces, plus `@byom-ai/ui` — a registry CLI that copies Tailwind-styled blocks into developer projects.

**Architecture:** Compound components (dot notation) with typed contexts, controlled/uncontrolled modes, forwardRef on all parts, data-* attributes for state. Primitives import from `@byom-ai/react` for hooks. Registry blocks are standalone .tsx files copied by CLI.

**Tech Stack:** React 18+, TypeScript (strict), Vitest, @testing-library/react, @byom-ai/react

**Spec:** `docs/superpowers/specs/2026-03-26-react-ui-design.md`

- [ ] **Step 1: Implement `create-context.ts`**

Typed context factory used by all compound components.

```ts
import { createContext, useContext } from "react";

export function createComponentContext<T>(componentName: string) {
  const Context = createContext<T | null>(null);
  Context.displayName = `${componentName}Context`;

  function useComponentContext(partName: string): T {
    const ctx = useContext(Context);
    if (ctx === null) {
      throw new Error(
        `<${partName}> must be used within <${componentName}>. ` +
        `Wrap it with <${componentName}> or <${componentName}.Root>.`,
      );
    }
    return ctx;
  }

  return [Context.Provider, useComponentContext] as const;
}
```

- [ ] **Step 2: Implement `forward-ref.ts`**

Typed forwardRef that preserves generic props and displayName.

```ts
import { forwardRef, type ForwardRefRenderFunction, type ComponentPropsWithRef } from "react";

export function createForwardRef<E extends HTMLElement, P = object>(
  displayName: string,
  render: ForwardRefRenderFunction<E, P>,
) {
  const Component = forwardRef<E, P>(render);
  Component.displayName = displayName;
  return Component;
}
```

- [ ] **Step 3: Implement `use-auto-scroll.ts`**

Auto-scroll to bottom when content changes. Uses RAF for throttling.

```ts
import { useRef, useEffect, useCallback } from "react";

export function useAutoScroll<E extends HTMLElement>(deps: unknown[]) {
  const ref = useRef<E>(null);
  const rafRef = useRef<number | null>(null);

  const scrollToBottom = useCallback(() => {
    if (rafRef.current !== null) return;
    rafRef.current = requestAnimationFrame(() => {
      if (ref.current) {
        ref.current.scrollTop = ref.current.scrollHeight;
      }
      rafRef.current = null;
    });
  }, []);

  useEffect(() => {
    scrollToBottom();
  }, deps); // eslint-disable-line react-hooks/exhaustive-deps

  useEffect(() => {
    return () => {
      if (rafRef.current !== null) cancelAnimationFrame(rafRef.current);
    };
  }, []);

  return { ref, scrollToBottom };
}
```
