# @arlopass/react Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build `@arlopass/react` — a React SDK wrapping `@arlopass/web-sdk` with hooks, providers, guards, error boundaries, and testing utilities.

**Architecture:** ClientStore adapts ArlopassClient for `useSyncExternalStore`. ArlopassProvider creates the client + store and shares via context. Hooks subscribe to store slices. Guards conditionally render based on store state. Testing utilities provide mock transport + mock provider.

**Tech Stack:** React 18+, TypeScript (strict), Vitest, @testing-library/react, @arlopass/web-sdk

**Spec:** `docs/superpowers/specs/2026-03-26-react-sdk-design.md`

```ts
import type { ArlopassTransport } from "@arlopass/web-sdk";

export function getInjectedTransport(): ArlopassTransport | null {
  if (
    typeof window !== "undefined" &&
    window.arlopass !== undefined &&
    typeof window.arlopass.request === "function"
  ) {
    return window.arlopass as ArlopassTransport;
  }
  return null;
}
```

```ts
// packages/react-sdk/src/store/snapshot.ts
import type { ClientState, ProviderDescriptor } from "../types.js";
import type { ArlopassSDKError } from "@arlopass/web-sdk";

export type ClientSnapshot = Readonly<{
  state: ClientState;
  sessionId: string | null;
  selectedProvider: Readonly<{ providerId: string; modelId: string }> | null;
  providers: readonly ProviderDescriptor[];
  error: ArlopassSDKError | null;
}>;

export type SnapshotInput = {
  state: ClientState;
  sessionId?: string | null;
  selectedProvider?: Readonly<{ providerId: string; modelId: string }> | null;
  providers?: readonly ProviderDescriptor[];
  error?: ArlopassSDKError | null;
};

export function createInitialSnapshot(): ClientSnapshot {
  return {
    state: "disconnected",
    sessionId: null,
    selectedProvider: null,
    providers: [],
    error: null,
  };
}

export function buildSnapshot(input: SnapshotInput): ClientSnapshot {
  return {
    state: input.state,
    sessionId: input.sessionId ?? null,
    selectedProvider: input.selectedProvider ?? null,
    providers: input.providers ?? [],
    error: input.error ?? null,
  };
}

export function snapshotsEqual(a: ClientSnapshot, b: ClientSnapshot): boolean {
  return (
    a.state === b.state &&
    a.sessionId === b.sessionId &&
    a.selectedProvider === b.selectedProvider &&
    a.providers === b.providers &&
    a.error === b.error
  );
}
```

```ts
// packages/react-sdk/src/store/subscriptions.ts
type Listener = () => void;

export class Subscriptions {
  #listeners = new Set<Listener>();

  subscribe(listener: Listener): () => void {
    this.#listeners.add(listener);
    return () => {
      this.#listeners.delete(listener);
    };
  }

  notify(): void {
    for (const listener of this.#listeners) {
      listener();
    }
  }

  clear(): void {
    this.#listeners.clear();
  }
}
```

```ts
// packages/react-sdk/src/store/client-store.ts
import type { ArlopassClient, ArlopassSDKError } from "@arlopass/web-sdk";
import type { ProviderDescriptor } from "../types.js";
import {
  buildSnapshot,
  createInitialSnapshot,
  snapshotsEqual,
  type ClientSnapshot,
} from "./snapshot.js";
import { Subscriptions } from "./subscriptions.js";

const HEARTBEAT_INTERVAL_MS = 500;

export class ClientStore {
  readonly #client: ArlopassClient;
  readonly #subscriptions = new Subscriptions();
  #snapshot: ClientSnapshot;
  #heartbeatId: ReturnType<typeof setInterval> | null = null;
  #providers: readonly ProviderDescriptor[] = [];
  #error: ArlopassSDKError | null = null;

  constructor(client: ArlopassClient) {
    this.#client = client;
    this.#snapshot = createInitialSnapshot();
    this.#startHeartbeat();
  }

  getSnapshot(): ClientSnapshot {
    return this.#snapshot;
  }

  subscribe(listener: () => void): () => void {
    return this.#subscriptions.subscribe(listener);
  }

  refreshSnapshot(): void {
    const next = buildSnapshot({
      state: this.#client.state,
      sessionId: this.#client.sessionId ?? null,
```

```tsx
"use client";

import { useEffect, useMemo, useRef, type ReactNode } from "react";
import { ArlopassClient } from "@arlopass/web-sdk";
import type { ArlopassSDKError } from "@arlopass/web-sdk";
import { ClientStore } from "../store/client-store.js";
import { getInjectedTransport } from "../transport/injected.js";
import { ArlopassContext, type ArlopassContextValue } from "./arlopass-context.js";
import type { ArlopassProviderProps } from "../types.js";

export function ArlopassProvider({
  appId,
  defaultProvider,
  defaultModel,
  autoConnect = true,
  onError,
  children,
}: ArlopassProviderProps): ReactNode {
  const storeRef = useRef<ClientStore | null>(null);
  const transportAvailableRef = useRef(false);

  // Create client + store once on mount
  if (storeRef.current === null) {
    const transport = getInjectedTransport();
    transportAvailableRef.current = transport !== null;

    if (transport !== null) {
      const client = new ArlopassClient({
        transport,
        origin: typeof window !== "undefined" ? window.location.origin : undefined,
      });
      storeRef.current = new ClientStore(client);
    } else {
      // Create a placeholder store with a dummy client that will fail on any operation
      // The transportAvailable flag prevents any operations from being attempted
      const dummyTransport = {
        async request() {
          throw new Error("Arlopass extension not installed.");
        },
        async stream() {
          throw new Error("Arlopass extension not installed.");
        },
      };
      const client = new ArlopassClient({ transport: dummyTransport });
      storeRef.current = new ClientStore(client);
    }
  }

  const store = storeRef.current;
  const transportAvailable = transportAvailableRef.current;

  // Auto-connect on mount
  useEffect(() => {
    if (!autoConnect || !transportAvailable || store === null) return;

    let cancelled = false;

    async function connectAndSelect(): Promise<void> {
      try {
        await store!.client.connect({ appId });
        store!.refreshSnapshot();

        if (
          !cancelled &&
          defaultProvider !== undefined &&
          defaultModel !== undefined
        ) {
          try {
            await store!.client.selectProvider({
              providerId: defaultProvider,
              modelId: defaultModel,
            });
            store!.refreshSnapshot();
          } catch (selectError) {
            store!.setError(selectError as ArlopassSDKError);
            onError?.(selectError as ArlopassSDKError);
          }
        }
      } catch (connectError) {
        if (!cancelled) {
          store!.setError(connectError as ArlopassSDKError);
          store!.refreshSnapshot();
          onError?.(connectError as ArlopassSDKError);
        }
      }
    }

    void connectAndSelect();

    return () => {
      cancelled = true;
    };
  }, [appId, autoConnect, defaultModel, defaultProvider, onError, store, transportAvailable]);

  // Cleanup on unmount
  useEffect(() => {
    return () => {
      if (store !== null) {
        void store.client.disconnect().catch(() => {});
        store.destroy();
      }
    };
  }, [store]);

  const contextValue = useMemo<ArlopassContextValue>(
    () => ({ store, transportAvailable }),
    [store, transportAvailable],
  );

  return (
    <ArlopassContext.Provider value={contextValue}>
      {children}
    </ArlopassContext.Provider>
  );
}
```

```ts
import { useCallback, useRef } from "react";
import type { ArlopassSDKError } from "@arlopass/web-sdk";
import type { ClientState } from "../types.js";
import { useArlopassContext, useStoreSnapshot } from "./use-store.js";

type UseConnectionReturn = Readonly<{
  state: ClientState;
  sessionId: string | null;
  isConnected: boolean;
  isConnecting: boolean;
  error: ArlopassSDKError | null;
  connect: () => Promise<void>;
  disconnect: () => Promise<void>;
  retry: (() => Promise<void>) | null;
}>;

export function useConnection(): UseConnectionReturn {
  const { store } = useArlopassContext();
  const snapshot = useStoreSnapshot();
  const lastConnectArgsRef = useRef<{ appId: string } | null>(null);

  const connect = useCallback(async () => {
    store.clearError();
    try {
      // appId is encoded in the client at construction time by the provider
      await store.client.connect({
        appId: "default", // Provider already configured the client
      });
      store.refreshSnapshot();
    } catch (error) {
      store.setError(error as ArlopassSDKError);
      store.refreshSnapshot();
      throw error;
    }
  }, [store]);

  const disconnect = useCallback(async () => {
    try {
      await store.client.disconnect();
    } finally {
      store.refreshSnapshot();
    }
  }, [store]);

  const retry = snapshot.error !== null && (snapshot.error as ArlopassSDKError).retryable === true
    ? async () => {
        store.clearError();
        await connect();
      }
    : null;

  return {
    state: snapshot.state,
    sessionId: snapshot.sessionId,
    isConnected: snapshot.state === "connected" || snapshot.state === "degraded",
    isConnecting: snapshot.state === "connecting" || snapshot.state === "reconnecting",
    error: snapshot.error,
    connect,
    disconnect,
    retry,
  };
}
```

```ts
// packages/react-sdk/src/hooks/use-providers.ts
"use client";

import { useCallback, useEffect, useRef, useState } from "react";
import type { ArlopassSDKError } from "@arlopass/web-sdk";
import type { ProviderDescriptor, SelectProviderInput } from "../types.js";
import type { SelectProviderResult } from "@arlopass/web-sdk";
import { useArlopassContext, useStoreSnapshot } from "./use-store.js";

type UseProvidersReturn = Readonly<{
  providers: readonly ProviderDescriptor[];
  selectedProvider: Readonly<{ providerId: string; modelId: string }> | null;
  isLoading: boolean;
  error: ArlopassSDKError | null;
  listProviders: () => Promise<readonly ProviderDescriptor[]>;
  selectProvider: (input: SelectProviderInput) => Promise<SelectProviderResult>;
  retry: (() => Promise<void>) | null;
}>;

export function useProviders(): UseProvidersReturn {
  const { store } = useArlopassContext();
  const snapshot = useStoreSnapshot();
  const [isLoading, setIsLoading] = useState(false);
  const [error, setError] = useState<ArlopassSDKError | null>(null);
  const hasFetchedRef = useRef(false);

  const listProviders = useCallback(async (): Promise<readonly ProviderDescriptor[]> => {
    setIsLoading(true);
    setError(null);
    try {
      const result = await store.client.listProviders();
      store.setProviders(result.providers);
      return result.providers;
    } catch (e) {
      const sdkError = e as ArlopassSDKError;
      setError(sdkError);
      throw e;
    } finally {
      setIsLoading(false);
    }
  }, [store]);

  const selectProvider = useCallback(
    async (input: SelectProviderInput): Promise<SelectProviderResult> => {
      setError(null);
      try {
        const result = await store.client.selectProvider(input);
        store.refreshSnapshot();
        return result;
      } catch (e) {
        const sdkError = e as ArlopassSDKError;
        setError(sdkError);
        throw e;
      }
    },
    [store],
  );

  // Auto-fetch when connected
  useEffect(() => {
    if (
      (snapshot.state === "connected" || snapshot.state === "degraded") &&
      !hasFetchedRef.current
    ) {
      hasFetchedRef.current = true;
      void listProviders().catch(() => {});
    }
    if (snapshot.state === "disconnected") {
      hasFetchedRef.current = false;
    }
  }, [snapshot.state, listProviders]);

  const retry =
    error !== null && (error as ArlopassSDKError).retryable === true
      ? async () => {
          setError(null);
          await listProviders();
        }
      : null;

  return {
    providers: snapshot.providers,
    selectedProvider: snapshot.selectedProvider,
    isLoading,
    error,
    listProviders,
    selectProvider,
    retry,
  };
}
```

```ts
function generateMessageId(): MessageId {
  if (typeof globalThis.crypto?.randomUUID === "function") {

  }
  const fallback = Array.from({ length: 16 }, () =>
    Math.floor(Math.random() * 16).toString(16),
  ).join("");
  return `msg.${fallback}`;
}

  initialMessages?: readonly ChatMessage[];
  onMessage?: (message: TrackedChatMessage) => void;
  onDelta?: (delta: string, accumulated: string, messageId: MessageId) => void;
  onError?: (error: ArlopassSDKError, messageId: MessageId) => void;
};

type UseChatReturn = Readonly<{

  isStreaming: boolean;

  stream: (content: string) => Promise<MessageId>;
  stop: () => void;
  clearMessages: () => void;

    isStreaming,
    isSending,

    stream,
```

```tsx
// packages/react-sdk/src/guards/arlopass-error-boundary.tsx
"use client";

import { Component, type ErrorInfo, type ReactNode } from "react";

type FallbackProps = {
  error: Error;
  resetErrorBoundary: () => void;
};

type Props = {
  fallback: (props: FallbackProps) => ReactNode;
  onError?: (error: Error, errorInfo: ErrorInfo) => void;
  children: ReactNode;
};

type State = {
  error: Error | null;
};

export class ArlopassErrorBoundary extends Component<Props, State> {
  constructor(props: Props) {
    super(props);
    this.state = { error: null };
  }

  static getDerivedStateFromError(error: Error): State {
    return { error };
  }

  componentDidCatch(error: Error, errorInfo: ErrorInfo): void {
    this.props.onError?.(error, errorInfo);
  }

  #resetErrorBoundary = (): void => {
    this.setState({ error: null });
  };

  render(): ReactNode {
    if (this.state.error !== null) {
      return this.props.fallback({
        error: this.state.error,
        resetErrorBoundary: this.#resetErrorBoundary,
      });
    }
    return this.props.children;
  }
}
```

```ts
// packages/react-sdk/src/testing/mock-transport.ts
import type { ArlopassTransport } from "@arlopass/web-sdk";
import type {
  TransportRequest,
  TransportResponse,
  TransportStream,
} from "@arlopass/web-sdk";

type MockTransportOptions = {
  providers?: Array<{
    providerId: string;
    providerName: string;
    models: string[];
  }>;
  chatResponse?: string | ((messages: Array<{ role: string; content: string }>) => string);
  streamResponse?: string;
  streamChunks?: string[];
  failOn?: { connect?: boolean; chat?: boolean; stream?: boolean };
  chatError?: Error;  // Accepts ArlopassSDKError or any Error subclass
  latency?: { connect?: number; chat?: number; stream?: number };
};

function delay(ms: number): Promise<void> {
  return new Promise((resolve) => setTimeout(resolve, ms));
}

export function createMockTransport(
  options: MockTransportOptions = {},
): ArlopassTransport {
  const providers = options.providers ?? [
    { providerId: "provider.mock", providerName: "Mock AI", models: ["model.test"] },
  ];

  return {
    async request<TReq, TRes>(
      request: TransportRequest<TReq>,
    ): Promise<TransportResponse<TRes>> {
      const capability = (request.envelope as Record<string, unknown>).capability as string;

      if (capability === "session.create" && options.failOn?.connect) {
        throw new Error("Mock: connection failed");
      }

      if (options.latency?.connect && capability === "session.create") {
        await delay(options.latency.connect);
      }

      if (capability === "chat.completions") {
        if (options.failOn?.chat) throw new Error("Mock: chat failed");
        if (options.chatError) throw options.chatError;
        if (options.latency?.chat) await delay(options.latency.chat);

        const payload = (request.envelope as Record<string, unknown>).payload as Record<string, unknown>;
        const messages = (payload?.messages ?? []) as Array<{ role: string; content: string }>;
        const responseText =
          typeof options.chatResponse === "function"
            ? options.chatResponse(messages)
            : (options.chatResponse ?? "Mock response");

        return {
          envelope: {
            ...request.envelope,
            payload: { message: { role: "assistant", content: responseText } },
          },
        } as unknown as TransportResponse<TRes>;
      }

      // Default: echo back with capabilities/providers
      return {
        envelope: {
          ...request.envelope,
          payload:
            capability === "provider.list"
              ? { providers }
              : { capabilities: ["chat.completions", "chat.stream", "provider.list", "session.create"] },
        },
      } as unknown as TransportResponse<TRes>;
    },

    async stream<TReq, TRes>(
      request: TransportRequest<TReq>,
    ): Promise<TransportStream<TRes>> {
      if (options.failOn?.stream) throw new Error("Mock: stream failed");

      const chunks =
        options.streamChunks ??
        (options.streamResponse ?? "Mock streamed response").split(" ").map((w, i, arr) =>
          i < arr.length - 1 ? w + " " : w,
        );

      const chunkDelay = options.latency?.stream ?? 0;

      async function* generate(): AsyncIterable<TransportResponse<TRes>> {
        let index = 0;
        for (const chunk of chunks) {
          if (chunkDelay > 0) await delay(chunkDelay);
          yield {
            envelope: {
              ...request.envelope,
              payload: { type: "chunk", delta: chunk, index: index++ },
            },
          } as unknown as TransportResponse<TRes>;
        }
        yield {
          envelope: {
            ...request.envelope,
            payload: { type: "done" },
          },
        } as unknown as TransportResponse<TRes>;
      }

      return generate();
    },

    async disconnect(): Promise<void> {},
  };
}
```

```ts
// packages/react-sdk/src/testing/test-helpers.ts
import type { ClientStore } from "../store/client-store.js";
import type { ClientSnapshot } from "../store/snapshot.js";

export async function waitForSnapshot(
  store: ClientStore,
  predicate: (snapshot: ClientSnapshot) => boolean,
  options?: { timeout?: number },
): Promise<ClientSnapshot> {
  const timeout = options?.timeout ?? 2000;
  const start = Date.now();

  return new Promise((resolve, reject) => {
    const check = (): void => {
      const snapshot = store.getSnapshot();
      if (predicate(snapshot)) {
        resolve(snapshot);
        return;
      }
      if (Date.now() - start > timeout) {
        reject(new Error(`waitForSnapshot timed out after ${timeout}ms`));
        return;
      }
      setTimeout(check, 50);
    };
    check();
  });
}

/**
 * Wait for streaming to finish. Resolves when `data-testid="arlopass-streaming"`
 * element disappears and `data-testid="arlopass-assistant-message"` appears.
 */
export async function waitForStream(
  screen: {
    findByTestId: (id: string) => Promise<HTMLElement>;
    queryByTestId: (id: string) => HTMLElement | null;
  },
  options?: { timeout?: number },
): Promise<string> {
  const timeout = options?.timeout ?? 5000;
  const start = Date.now();

  return new Promise((resolve, reject) => {
    const check = (): void => {
      const streaming = screen.queryByTestId("arlopass-streaming");
      const message = screen.queryByTestId("arlopass-assistant-message");
      if (streaming === null && message !== null) {
        resolve(message.textContent ?? "");
        return;
      }
      if (Date.now() - start > timeout) {
        reject(new Error(`waitForStream timed out after ${timeout}ms`));
        return;
      }
      setTimeout(check, 50);
    };
    check();
  });
}
```
